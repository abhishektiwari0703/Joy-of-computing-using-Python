def countStrings(n):
    A=[0 for a in range(n)]
    B=[0 for a in range(n)]
    A[0] = B[0] = 1
    for a in range(1,n):
        A[a] = A[a-1] + B[a-1]
        B[a] = A[a-1]
    return(A[n-1] + B[n-1])
n=int(input())
if n>0:
  print(countStrings(n),end="")
else:
  print("invalid",end="")
