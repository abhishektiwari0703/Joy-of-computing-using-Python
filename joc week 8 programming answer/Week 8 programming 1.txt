M,N= input().split()
M,N=int(M),int(N)
mat,check=[],0
for  i in range(M):
    mat.append(input().split())
for x in mat:
    for y in x:
        if y=="0":
            check=check+1
        elif y=="1":
            check=check+1
if check==M*N:
    print("Yes",end="")
else:
    print("No",end="")
